# Example: Scraping YouTube Comments
from googleapiclient.discovery import build
import pandas as pd

api_key = "YOUR_API_KEY"
video_id = "VIDEO_ID_HERE"

youtube = build("youtube", "v3", developerKey=api_key)
comments = []

request = youtube.commentThreads().list(
    part="snippet",
    videoId=video_id,
    maxResults=100,
    textFormat="plainText"
)
response = request.execute()

for item in response["items"]:
    comment = item["snippet"]["topLevelComment"]["snippet"]["textDisplay"]
    comments.append(comment)

df = pd.DataFrame(comments, columns=["Comment"])
df.to_csv("youtube_comments.csv", index=False)
print("âœ… Saved to youtube_comments.csv")

Extracts comments from a YouTube video using the official API for further sentiment or framing analysis.

# Example: Collecting trending TikTok videos
import os
import asyncio
from TikTokApi import TikTokApi

ms_token = os.environ.get("ms_token", None)

async def test_trending():
    async with TikTokApi() as api:
        await api.create_sessions(ms_tokens=[ms_token], num_sessions=1, sleep_after=3, browser="chromium")
        async for video in api.trending.videos(count=10):
            data = video.as_dict
            print(f"{data['id']} | {data['desc']} | by @{data['author']['uniqueId']}")

if __name__ == "__main__":
    asyncio.run(test_trending())

Collects metadata from TikTok trending videos, including captions and authors â€” useful for identifying emerging digital narratives.

# Example: Scraping News Headlines
import requests
from bs4 import BeautifulSoup

url = "https://www.bbc.com/news"
soup = BeautifulSoup(requests.get(url).text, "html.parser")

headlines = [h3.text for h3 in soup.find_all("h3")]
print(headlines[:10])

Collects recent news headlines to map dominant narratives and policy-related discussions in mainstream media.
